45  template <class Element>
46  class ThreadedPipeline : public Pipeline<Element>
47  {
48  private:
49  threadpp::ThreadGroup mThreads;
50  threadpp::Mutex mMutex;
51  threadpp::Condition mCond, mMainCond;
52 
53  bool mThreadsDone;
54  unsigned mWorkingThreads;
55  PipelineJobQueue mJobsDone;
56  void threadFunction ()
57  {
58  Cache *cache = NULL;
59  threadpp::Mutex::Lock lk(mMutex);
60  while (!mThreadsDone)
61  {
62  if (Pipeline<Element>::mJobs.empty())
63  mCond.wait(lk);
64  if (!Pipeline<Element>::mJobs.empty())
65  {
66  PipelineJob *job = Pipeline<Element>::mJobs.front ();
67  Pipeline<Element>::mJobs.pop ();
68  ++mWorkingThreads;
69  lk.unlock ();
70  if (!cache)
71  cache = Pipeline<Element>::createCache();
72  job->execute(cache);
73  lk.lock ();
74  --mWorkingThreads;
75  mJobsDone.push (job);
76  mMainCond.notifyOne ();
77  }
78  }
79  lk.unlock ();
80  if (cache)
81  {
82  Pipeline<Element>::freeCache (cache);
83  }
84  }
85  static void *threadEntry (void *pipe)
86  {
87  (static_cast<ThreadedPipeline<Element>*>(pipe))->threadFunction ();
88  return NULL;
89  }
90 
91  public:
94  ThreadedPipeline (size_t numberOfThreads) : mThreadsDone(false), mWorkingThreads(0)
95  {
96  NoiseAssert (numberOfThreads > 0, numberOfThreads);
97  for (size_t i=0;i<numberOfThreads;++i)
98  {
99  mThreads.createThread (threadEntry, this);
100  }
101  }
104  virtual void executeJobs ()
105  {
106  mCond.notifyAll();
107  threadpp::Mutex::Lock lk(mMutex);
108  while (!Pipeline<Element>::mJobs.empty() || mWorkingThreads > 0)
109  {
110  if (!Pipeline<Element>::mJobs.empty() || mWorkingThreads > 0)
111  mMainCond.wait(lk);
112  while (!mJobsDone.empty())
113  {
114  PipelineJob *job = mJobsDone.front ();
115  mJobsDone.pop ();
116  lk.unlock ();
117  job->finish ();
118  delete job;
119  lk.lock ();
120  }
121  }
122  }
124  virtual void addJob (PipelineJob *job)
125  {
126  NoiseAssert (job != NULL, job);
127  threadpp::Mutex::Lock lk(mMutex);
128  Pipeline<Element>::mJobs.push (job);
129  }
131  virtual ~ThreadedPipeline ()
132  {
133  mThreadsDone = true;
134  mCond.notifyAll();
135  mThreads.join ();
136  }
137  };
