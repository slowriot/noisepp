34  class ExponentElement1D : public PipelineElement1D
35  {
36  private:
37  ElementID mElement;
38  Real mExponent;
39  const PipelineElement1D *mElementPtr;
40 
41  public:
42  ExponentElement1D (const Pipeline1D *pipe, ElementID element, Real exponent) : mElement(element), mExponent(exponent)
43  {
44  mElementPtr = pipe->getElement(element);
45  }
46  virtual Real getValue (Real x, Cache *cache) const
47  {
48  Real value;
49  value = getElementValue (mElementPtr, mElement, x, cache);
50  return (std::pow (std::fabs ((value + Real(1.0)) / Real(2.0)), mExponent) * Real(2.0) - Real(1.0));
51  }
