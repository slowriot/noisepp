35  class BillowElement1D : public PipelineElement1D
36  {
37  private:
38  struct Octave
39  {
40  int seed;
41  Real scale;
42  Real persistence;
43  };
44  Octave *mOctaves;
45  size_t mOctaveCount;
46  int mQuality;

49  NOISEPP_INLINE Real calculateGradient (Real x, int seed) const
50  {
51  if (mQuality == NOISE_QUALITY_STD)
52  return Generator1D::calcGradientCoherentNoiseStd (x, seed, mScale);
53  else if (mQuality == NOISE_QUALITY_HIGH)
54  return Generator1D::calcGradientCoherentNoiseHigh (x, seed, mScale);
55  else if (mQuality == NOISE_QUALITY_LOW)
56  return Generator1D::calcGradientCoherentNoiseLow (x, seed, mScale);
57  else if (mQuality == NOISE_QUALITY_FAST_STD)
58  return Generator1D::calcGradientCoherentFastNoiseStd (x, seed, mScale);
59  else if (mQuality == NOISE_QUALITY_FAST_HIGH)
60  return Generator1D::calcGradientCoherentFastNoiseHigh (x, seed, mScale);
61  else
62  return Generator1D::calcGradientCoherentFastNoiseLow (x, seed, mScale);
63  }
64  public:
65  BillowElement1D (size_t octaves, Real frequency, Real lacunarity, Real persistence, int mainSeed, int quality, Real nscale) : mOctaveCount(octaves), mQuality(quality), mScale(nscale)
66  {
67  if (quality > NOISE_QUALITY_HIGH)
68  mScale *= FAST_NOISE_SCALE_FACTOR;
69  mOctaves = new Octave[mOctaveCount];
70  Real curPersistence = 1.0;
71  int seed;
72  Real scale = frequency;
73  for (size_t o=0;o<mOctaveCount;++o)
74  {
75  seed = (mainSeed + int(o)) & 0xffffffff;
76  mOctaves[o].persistence = curPersistence;
77  mOctaves[o].scale = scale;
78  mOctaves[o].seed = seed;
79 
80  scale *= lacunarity;
81  curPersistence *= persistence;
82  }
83  }
84  virtual ~BillowElement1D ()
85  {
86  delete mOctaves;
87  mOctaves = NULL;
88  }
89  virtual Real getValue (Real x, Cache *cache) const
90  {
91  Real value = 0.5;
92  Real signal = 1.0;
93 
94  for (size_t o=0;o<mOctaveCount;++o)
95  {
96  const Real nx = Math::MakeInt32Range (x * mOctaves[o].scale);
97  signal = calculateGradient(nx, mOctaves[o].seed);
98  signal = Real(2.0) * std::fabs (signal) - Real(1.0);
99 
100  value += signal * mOctaves[o].persistence;
101  }
102 
103  return value;
104  }
105  };
