139  class PerlinElement1D : public PipelineElement1D
140  {
141  private:
142  struct Octave
143  {
144  int seed;
145  Real scale;
146  Real persistence;
147  };
148  Octave *mOctaves;
149  size_t mOctaveCount;
150  int mQuality;
151  Real mScale;
152 
153  NOISEPP_INLINE Real calculateGradient (Real x, int seed) const
154  {
155  if (mQuality == NOISE_QUALITY_STD)
156  return Generator1D::calcGradientCoherentNoiseStd (x, seed, mScale);
157  else if (mQuality == NOISE_QUALITY_HIGH)
158  return Generator1D::calcGradientCoherentNoiseHigh (x, seed, mScale);
159  else if (mQuality == NOISE_QUALITY_LOW)
160  return Generator1D::calcGradientCoherentNoiseLow (x, seed, mScale);
161  else if (mQuality == NOISE_QUALITY_FAST_STD)
162  return Generator1D::calcGradientCoherentFastNoiseStd (x, seed, mScale);
163  else if (mQuality == NOISE_QUALITY_FAST_HIGH)
164  return Generator1D::calcGradientCoherentFastNoiseHigh (x, seed, mScale);
165  else
166  return Generator1D::calcGradientCoherentFastNoiseLow (x, seed, mScale);
167  }
168  public:
169  PerlinElement1D (size_t octaves, Real frequency, Real lacunarity, Real persistence, int mainSeed, int quality, Real nscale) : mOctaveCount(octaves), mQuality(quality), mScale(nscale)
170  {
171  if (quality > NOISE_QUALITY_HIGH)
172  mScale *= FAST_NOISE_SCALE_FACTOR;
173  mOctaves = new Octave[mOctaveCount];
174  Real curPersistence = 1.0;
175  int seed;
176  Real scale = frequency;
177  for (size_t o=0;o<mOctaveCount;++o)
178  {
179  seed = (mainSeed + int(o)) & 0xffffffff;
180  mOctaves[o].persistence = curPersistence;
181  mOctaves[o].scale = scale;
182  mOctaves[o].seed = seed;
183 
184  scale *= lacunarity;
185  curPersistence *= persistence;
186  }
187  }
188  virtual ~PerlinElement1D ()
189  {
190  delete mOctaves;
191  mOctaves = NULL;
192  }
193  virtual Real getValue (Real x, Cache *cache) const
194  {
195  Real value = 0.0;
196  Real signal = 1.0;
197 
198  for (size_t o=0;o<mOctaveCount;++o)
199  {
200  const Real nx = Math::MakeInt32Range (x * mOctaves[o].scale);
201  signal = calculateGradient(nx, mOctaves[o].seed);
202 
203  value += signal * mOctaves[o].persistence;
204  }
205 
206  return value;
207  }
208  };
