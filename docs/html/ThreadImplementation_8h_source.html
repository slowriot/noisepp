31  class Thread
32  {
33  private:
34  bool mJoinable;
35 #if THREADPP_PLATFORM == THREADPP_PLATFORM_UNIX
36  pthread_t mThread;
37 #elif THREADPP_PLATFORM == THREADPP_PLATFORM_WINDOWS
38  HANDLE mThreadHandle;
39  DWORD mThreadId;
40 #endif
41  public:
43  THREADPP_INLINE Thread (void *(*start_routine)(void *), void *arg=NULL) : mJoinable(true)
44  {
45 #if THREADPP_PLATFORM == THREADPP_PLATFORM_UNIX
46  pthread_create(&mThread, NULL, start_routine, arg);
47 #elif THREADPP_PLATFORM == THREADPP_PLATFORM_WINDOWS
48  mThreadHandle = CreateThread (0, 0, (LPTHREAD_START_ROUTINE)start_routine, arg, 0, &mThreadId);
49 #endif
50  }
51  THREADPP_INLINE ~Thread ()
52  {
53  if (mJoinable)
54  {
55 #if THREADPP_PLATFORM == THREADPP_PLATFORM_UNIX
56  pthread_detach(mThread);
57 #elif THREADPP_PLATFORM == THREADPP_PLATFORM_WINDOWS
58  CloseHandle (mThreadHandle);
59 #endif
60  }
61  }
63  THREADPP_INLINE void join ()
64  {
65  assert (mJoinable);
66 #if THREADPP_PLATFORM == THREADPP_PLATFORM_UNIX
67  pthread_join (mThread, NULL);
68 #elif THREADPP_PLATFORM == THREADPP_PLATFORM_WINDOWS
69  WaitForSingleObject (mThreadHandle, INFINITE);
70  CloseHandle (mThreadHandle);
71 #endif
72  mJoinable = false;
73  }
74  };
