50  template <class PipelineElement>
51  class CurveElementBase : public PipelineElement
52  {
53  protected:
54  ElementID mElement;
55  const PipelineElement *mElementPtr;
56  CurveControlPoint *mControlPoints;
57  int mControlPointCount;
58 
59  NOISEPP_INLINE int clampValue (int i, int lo, int hi) const
60  {
61  if (i < lo)
62  return lo;
63  else if (i > hi)
64  return hi;
65  else
66  return i;
67  }
68 
69  NOISEPP_INLINE Real mapValue (Real value) const
70  {
71  int index;
72  for (index=0;index<mControlPointCount;++index)
73  {
74  if (value < mControlPoints[index].inValue)
75  {
76  break;
77  }
78  }
79 
80  const int index0 = clampValue (index-2, 0, mControlPointCount-1);
81  const int index1 = clampValue (index-1, 0, mControlPointCount-1);
82  const int index2 = clampValue (index, 0, mControlPointCount-1);
83  const int index3 = clampValue (index+1, 0, mControlPointCount-1);
84 
85  if (index1 == index2)
86  {
87  return mControlPoints[index1].outValue;
88  }
89 
90  const Real in0 = mControlPoints[index1].inValue;
91  const Real in1 = mControlPoints[index2].inValue;
92  const Real a = (value - in0) / (in1 - in0);
93  return Math::InterpCubic (mControlPoints[index0].outValue, mControlPoints[index1].outValue, mControlPoints[index2].outValue, mControlPoints[index3].outValue, a);
94  }
95  public:
96  CurveElementBase (const Pipeline<PipelineElement> *pipe, ElementID element, CurveControlPoint *points, int count) : mElement(element), mControlPoints(points), mControlPointCount(count)
97  {
98  NoiseAssert (points != NULL, points);
99  NoiseAssert (count >= 4, count);
100  mElementPtr = pipe->getElement (mElement);
101  }
102  virtual ~CurveElementBase ()
103  {
104  delete[] mControlPoints;
105  mControlPoints = NULL;
106  }
107  };
