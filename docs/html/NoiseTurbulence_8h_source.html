34  {
35  private:

38  const PipelineElement1D *mElementPtr;
39  ElementID mPerlinX;

41 
42  public:
43  TurbulenceElement1D (const Pipeline1D *pipe, Real power, ElementID element, ElementID perlinX) :
44  mPower(power), mElement(element), mPerlinX(perlinX)

93  class TurbulenceElement3D : public PipelineElement3D
94  {
95  private:
96  Real mPower;
97  ElementID mElement;
98  const PipelineElement3D *mElementPtr;
99  ElementID mPerlinX;
100  const PipelineElement3D *mPerlinXPtr;
101  ElementID mPerlinY;
102  const PipelineElement3D *mPerlinYPtr;
103  ElementID mPerlinZ;
104  const PipelineElement3D *mPerlinZPtr;
105 
106  public:
107  TurbulenceElement3D (const Pipeline3D *pipe, Real power, ElementID element, ElementID perlinX, ElementID perlinY, ElementID perlinZ) :
108  mPower(power), mElement(element), mPerlinX(perlinX), mPerlinY(perlinY), mPerlinZ(perlinZ)
109  {
110  mElementPtr = pipe->getElement (mElement);
111  mPerlinXPtr = pipe->getElement (mPerlinX);
112  mPerlinYPtr = pipe->getElement (mPerlinY);
113  mPerlinZPtr = pipe->getElement (mPerlinZ);
114  }
115  virtual Real getValue (Real x, Real y, Real z, Cache *cache) const
116  {
117  Real x0, y0, z0;
118  Real x1, y1, z1;
119  Real x2, y2, z2;
120  x0 = x + Real(12414.0 / 65536.0);
121  y0 = y + Real(65124.0 / 65536.0);
122  z0 = z + Real(31337.0 / 65536.0);
123  x1 = x + Real(26519.0 / 65536.0);
124  y1 = y + Real(18128.0 / 65536.0);
125  z1 = z + Real(60493.0 / 65536.0);
126  x2 = x + Real(53820.0 / 65536.0);
127  y2 = y + Real(11213.0 / 65536.0);
128  z2 = z + Real(44845.0 / 65536.0);
129  Real xFinal = x + (getElementValue (mPerlinXPtr, mPerlinX, x0, y0, z0, cache) * mPower);
130  Real yFinal = y + (getElementValue (mPerlinYPtr, mPerlinY, x1, y1, z1, cache) * mPower);
131  Real zFinal = z + (getElementValue (mPerlinZPtr, mPerlinZ, x2, y2, z2, cache) * mPower);
132  return getElementValue (mElementPtr, mElement, xFinal, yFinal, zFinal, cache);
133  }

135  };
