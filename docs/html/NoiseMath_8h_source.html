45  {
46  const Real x = v3 - v2 - v0 + v1;

48  }

55  static NOISEPP_INLINE Real CubicCurve5 (Real a)
56  {
57  const Real a3 = a * a * a;
58  const Real a4 = a3 * a;
59  const Real a5 = a4 * a;
60  return Real(10) * a3 - Real(15) * a4 + Real(6) * a5;
61  }
63  static NOISEPP_INLINE Real MakeInt32Range (Real n)
64  {
65  if (n >= Real(1073741824.0))
66  {
67  Real temp(1073741824.0);
68  return (Real(2.0) * std::modf (n, &temp)) - Real(1073741824.0);
69  }
70  else if (n <= Real(-1073741824.0))
71  {
72  Real temp(1073741824.0);
73  return (Real(2.0) * std::modf (n, &temp)) + Real(1073741824.0);
74  }
75  else
76  return n;
77  }
