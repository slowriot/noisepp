36  class VoronoiElement2D : public PipelineElement2D
37  {
38  private:
39  Real mFrequency;
40  int mSeed;
41  Real mDisplacement;
42  bool mEnableDistance;
43 
44  public:
45  VoronoiElement2D (Real frequency, int seed, Real displacement, bool enableDistance) : mFrequency(frequency), mSeed(seed), mDisplacement(displacement), mEnableDistance(enableDistance)
46  {
47  }
48  virtual Real getValue (Real x, Real y, Cache *cache) const
49  {
50  x *= mFrequency;
51  y *= mFrequency;
52 
53  int xi = (x > Real(0.0) ? (int)x : (int)x - 1);
54  int yi = (y > Real(0.0) ? (int)y : (int)y - 1);
55 
56  Real minDist = Real(2147483647.0);
57  Real xmin = Real(0);
58  Real ymin = Real(0);
59 
60  for (int xc=xi-2;xc<=xi+2;++xc)
61  {
62  for (int yc=yi-2;yc<=yi+2;++yc)
63  {
64  Real xp = xc + Generator2D::calcNoise(xc, yc, mSeed);
65  Real yp = yc + Generator2D::calcNoise(xc, yc, mSeed+1);
66  Real xd = xp - x;
67  Real yd = yp - y;
68  Real dist = xd * xd + yd * yd;
69  if (dist < minDist)
70  {
71  minDist = dist;
72  xmin = xp;
73  ymin = yp;
74  }
75  }
76  }
77 
78  Real value;
79  if (mEnableDistance)
80  {
81  const Real SQRT_TWO = 1.4142135623730951;
82  Real xDist = xmin - x;
83  Real yDist = ymin - y;
84  value = (std::sqrt(xDist * xDist + yDist * yDist)) * SQRT_TWO - Real(1.0);
85  }
86  else
87  {
88  value = Real(0.0);
89  }
90 
91  return value + (mDisplacement * (Real)Generator2D::calcNoise((int)floor(xmin), (int)floor(ymin)));
92  }
93  };

106  }
107  virtual Real getValue (Real x, Real y, Real z, Cache *cache) const
108  {
109  x *= mFrequency;
110  y *= mFrequency;
111  z *= mFrequency;
112 
113  int xi = (x > Real(0.0) ? (int)x : (int)x - 1);
114  int yi = (y > Real(0.0) ? (int)y : (int)y - 1);
115  int zi = (z > Real(0.0) ? (int)z : (int)z - 1);
116 
117  Real minDist = Real(2147483647.0);
118  Real xmin = Real(0);
119  Real ymin = Real(0);
120  Real zmin = Real(0);
121 
122  for (int xc=xi-2;xc<=xi+2;++xc)
123  {
124  for (int yc=yi-2;yc<=yi+2;++yc)
125  {
126  for (int zc=zi-2;zc<=zi+2;++zc)
127  {
128  Real xp = xc + Generator3D::calcNoise(xc, yc, zc, mSeed);
129  Real yp = yc + Generator3D::calcNoise(xc, yc, zc, mSeed+1);
130  Real zp = zc + Generator3D::calcNoise(xc, yc, zc, mSeed+2);
131  Real xd = xp - x;
132  Real yd = yp - y;
133  Real zd = zp - z;
134  Real dist = xd * xd + yd * yd + zd * zd;
135  if (dist < minDist)
136  {
137  minDist = dist;
138  xmin = xp;
139  ymin = yp;
140  zmin = zp;
141  }
142  }
143  }
144  }
145 
146  Real value;
147  if (mEnableDistance)
148  {
149  const Real SQRT_THREE = 1.7320508075688772;
150  Real xDist = xmin - x;
151  Real yDist = ymin - y;
152  Real zDist = zmin - z;
153  value = (std::sqrt(xDist * xDist + yDist * yDist + zDist * zDist)) * SQRT_THREE - Real(1.0);
154  }
155  else
156  {
157  value = Real(0.0);
158  }
159 
160  return value + (mDisplacement * (Real)Generator3D::calcNoise((int)floor(xmin), (int)floor(ymin), (int)floor(zmin)));
161  }
