138  ElementID addElement (const Module *parent, Element *element)
139  {
140  NoiseAssert (element != NULL, element);
141  NoiseAssert (parent != NULL, parent);
142  std::map<const Module*, ElementID>::iterator it = mElementIDs.find(parent);
143  if (it != mElementIDs.end())

145  delete element;

173  }

180  virtual void addJob (PipelineJob *job)
181  {
182  NoiseAssert (job != NULL, job);
183  mJobs.push (job);
184  }
186  virtual void executeJobs ()
187  {
188  Cache *cache = createCache();
189  PipelineJob *job;
190  while (!mJobs.empty())
191  {
192  job = mJobs.front ();
193  mJobs.pop ();
194  job->execute(cache);
195  job->finish();
196  delete job;
197  }
198  freeCache (cache);
199  }
201  virtual ~Pipeline ()
202  {
203  typename std::vector<Element*>::iterator itEnd = mElements.end();
204  for (typename std::vector<Element*>::iterator it=mElements.begin();it!=itEnd;++it)
205  {
206  delete *it;
207  }
208  mElements.clear ();
209  mElementIDs.clear ();
210  while (!mJobs.empty())
211  {
212  delete mJobs.front ();
213  mJobs.pop ();
214  }
215  }
216  };
