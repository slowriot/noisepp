36  class RidgedMultiElement1D : public PipelineElement1D
37  {
38  private:
39  struct Octave
40  {
41  int seed;
42  Real scale;
43  Real spectralWeight;
44  };
45  Octave *mOctaves;
46  size_t mOctaveCount;
47  int mQuality;
48  Real mOffset;
49  Real mGain;
50  Real mScale;

52  NOISEPP_INLINE Real calculateGradient (Real x, int seed) const
53  {
54  if (mQuality == NOISE_QUALITY_STD)
55  return Generator1D::calcGradientCoherentNoiseStd (x, seed, mScale);
56  else if (mQuality == NOISE_QUALITY_HIGH)
57  return Generator1D::calcGradientCoherentNoiseHigh (x, seed, mScale);
58  else if (mQuality == NOISE_QUALITY_LOW)
59  return Generator1D::calcGradientCoherentNoiseLow (x, seed, mScale);
60  else if (mQuality == NOISE_QUALITY_FAST_STD)
61  return Generator1D::calcGradientCoherentFastNoiseStd (x, seed, mScale);
62  else if (mQuality == NOISE_QUALITY_FAST_HIGH)
63  return Generator1D::calcGradientCoherentFastNoiseHigh (x, seed, mScale);
64  else
65  return Generator1D::calcGradientCoherentFastNoiseLow (x, seed, mScale);
66  }
67  public:
68  RidgedMultiElement1D (size_t octaves, Real frequency, Real lacunarity, Real exponent, Real offset, Real gain, int mainSeed, int quality, Real nscale) : mOctaveCount(octaves), mQuality(quality), mOffset(offset), mGain(gain), mScale(nscale)
69  {
70  if (quality > NOISE_QUALITY_HIGH)
71  mScale *= FAST_NOISE_SCALE_FACTOR;
72  mOctaves = new Octave[mOctaveCount];
73  int seed;
74  Real scale = frequency;
75  Real sw_freq = 1.0;
76  for (size_t o=0;o<mOctaveCount;++o)
77  {
78  seed = (mainSeed + int(o)) & 0x7fffffff;
79  mOctaves[o].spectralWeight = pow(sw_freq, -exponent);
80  mOctaves[o].scale = scale;
81  mOctaves[o].seed = seed;
82 
83  scale *= lacunarity;
84  sw_freq *= lacunarity;
85  }
86  }
87  virtual ~RidgedMultiElement1D ()
88  {
89  delete mOctaves;
90  mOctaves = NULL;
91  }
92  virtual Real getValue (Real x, Cache *cache) const
93  {
94  Real value = 0.0;
95  Real signal = 0.0;
96  Real weight = 1.0;
97 
98  for (size_t o=0;o<mOctaveCount;++o)
99  {
100  const Real nx = Math::MakeInt32Range (x * mOctaves[o].scale);
101  signal = calculateGradient(nx, mOctaves[o].seed);
102  signal = mOffset - std::fabs(signal);
103  signal *= signal;
104  signal *= weight;
105  weight = signal * mGain;
106  if (weight > Real(1.0))
107  weight = Real(1.0);
108  if (weight < Real(-1.0))
109  weight = Real(-1.0);
110 
111  value += signal * mOctaves[o].spectralWeight;
112  }
113 
114  return (value * Real(1.25)) - Real(1.0);
115  }
116  };
