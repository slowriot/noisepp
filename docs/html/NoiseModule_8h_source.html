71  class Module
72  {
73  protected:
75  const Module **mSourceModules;
77  size_t mSourceModuleCount;
78 
79  public:
81  Module (size_t sourceModuleCount=0) : mSourceModules(NULL), mSourceModuleCount(sourceModuleCount)
82  {
83  if (mSourceModuleCount)
84  {
85  mSourceModules = new const Module *[mSourceModuleCount];
86  memset (mSourceModules, 0, mSourceModuleCount*sizeof(Module*));
87  }
88  }
90  size_t getSourceModuleCount () const
91  {
92  return mSourceModuleCount;
93  }
96  bool walkTree (const Module *module) const
97  {
98  for (size_t i=0;i<mSourceModuleCount;++i)
99  {
100  if (mSourceModules[i] == module)
101  return true;
102  }
103  for (size_t i=0;i<mSourceModuleCount;++i)
104  {
105  if (mSourceModules[i] && mSourceModules[i]->walkTree(module))
106  return true;
107  }
108  return false;
109  }
113  void setSourceModule (size_t id, const Module *module)
114  {
115  NoiseAssertRange (id, mSourceModuleCount);
116  NoiseAssert (module != NULL, module);
117  NoiseAssert (module != this, module);
118  NoiseAssert (!module->walkTree(this), module);
119  mSourceModules[id] = module;
120  }
122  void setSourceModule (size_t id, const Module &module)
123  {
124  setSourceModule (id, &module);
125  }
127  const Module *getSourceModule (size_t id) const
128  {
129  NoiseAssertRange (id, mSourceModuleCount);
130  return mSourceModules[id];
131  }
133  virtual ElementID addToPipeline (Pipeline1D *pipe) const = 0;
135  virtual ElementID addToPipeline (Pipeline2D *pipe) const = 0;
137  virtual ElementID addToPipeline (Pipeline3D *pipe) const = 0;
139  ElementID addToPipe (Pipeline1D &pipe) const
140  {
141  return addToPipeline(&pipe);
142  }
144  ElementID addToPipe (Pipeline2D &pipe) const
145  {
146  return addToPipeline(&pipe);
147  }
149  ElementID addToPipe (Pipeline3D &pipe) const
150  {
151  return addToPipeline(&pipe);
152  }
154  virtual ModuleTypeId getType() const = 0;
155  virtual ~Module ()
156  {
157  if (mSourceModules)
158  {
159  delete[] mSourceModules;
160  mSourceModules = NULL;
161  }
162  }
163 #if NOISEPP_ENABLE_UTILS
164  virtual void write (utils::OutStream &stream) const;
167  virtual void read (utils::InStream &stream);
168 #endif
169  };
