56  class CheckerboardElement3D : public PipelineElement3D
57  {
58  public:
59  virtual Real getValue (Real x, Real y, Real z, Cache *cache) const
60  {
61  const int ix = (int)(floor (Math::MakeInt32Range (x)));
62  const int iy = (int)(floor (Math::MakeInt32Range (y)));
63  const int iz = (int)(floor (Math::MakeInt32Range (z)));
64  return ((ix & 1) ^ (iy & 1) ^ (iz & 1))? Real(-1.0) : Real(1.0);
65  }
